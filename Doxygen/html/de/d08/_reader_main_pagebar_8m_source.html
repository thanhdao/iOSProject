81 - (void)updatePageThumbView:(NSInteger)page
82 {
83  NSInteger pages = [document.pageCount integerValue];
84 
85  if (pages > 1) // Only update frame if more than one page
86  {
87  CGFloat controlWidth = trackControl.bounds.size.width;
88 
89  CGFloat useableWidth = (controlWidth - THUMB_LARGE_WIDTH);
90 
91  CGFloat stride = (useableWidth / (pages - 1)); // Page stride
92 
93  NSInteger X = (stride * (page - 1)); CGFloat pageThumbX = X;
94 
95  CGRect pageThumbRect = pageThumbView.frame; // Current frame
96 
97  if (pageThumbX != pageThumbRect.origin.x) // Only if different
98  {
99  pageThumbRect.origin.x = pageThumbX; // The new X position
100 
101  pageThumbView.frame = pageThumbRect; // Update the frame
102  }
103  }
104 
105  if (page != pageThumbView.tag) // Only if page number changed
106  {
107  pageThumbView.tag = page; [pageThumbView reuse]; // Reuse the thumb view
108 
109  CGSize size = CGSizeMake(THUMB_LARGE_WIDTH, THUMB_LARGE_HEIGHT); // Maximum thumb size
110 
111  NSURL *fileURL = document.fileURL; NSString *guid = document.guid; NSString *phrase = document.password;
112 
113  ReaderThumbRequest *request = [ReaderThumbRequest newForView:pageThumbView fileURL:fileURL password:phrase guid:guid page:page size:size];
114 
115  UIImage *image = [[ReaderThumbCache sharedInstance] thumbRequest:request priority:YES]; // Request the thumb
116

221 - (void)layoutSubviews
222 {
223  CGRect controlRect = CGRectInset(self.bounds, 4.0f, 0.0f);
224 
225  CGFloat thumbWidth = (THUMB_SMALL_WIDTH + THUMB_SMALL_GAP);
226 
227  NSInteger thumbs = (controlRect.size.width / thumbWidth);
228 
229  NSInteger pages = [document.pageCount integerValue]; // Pages
230 
231  if (thumbs > pages) thumbs = pages; // No more than total pages
232 
233  CGFloat controlWidth = ((thumbs * thumbWidth) - THUMB_SMALL_GAP);
234 
235  controlRect.size.width = controlWidth; // Update control width
236 
237  CGFloat widthDelta = (self.bounds.size.width - controlWidth);
238 
239  NSInteger X = (widthDelta / 2.0f); controlRect.origin.x = X;
240 
241  trackControl.frame = controlRect; // Update track control frame
242 
243  if (pageThumbView == nil) // Create the page thumb view when needed
244  {
245  CGFloat heightDelta = (controlRect.size.height - THUMB_LARGE_HEIGHT);
246 
247  NSInteger thumbY = (heightDelta / 2.0f); NSInteger thumbX = 0; // Thumb X, Y
248 
249  CGRect thumbRect = CGRectMake(thumbX, thumbY, THUMB_LARGE_WIDTH, THUMB_LARGE_HEIGHT);
250 
251  pageThumbView = [[ReaderPagebarThumb alloc] initWithFrame:thumbRect]; // Create the thumb view
252 
253  pageThumbView.layer.zPosition = 1.0f; // Z position so that it sits on top of the small thumbs
254 
255  [trackControl addSubview:pageThumbView]; // Add as the first subview of the track control
256  }
257 
258  [self updatePageThumbView:[document.pageNumber integerValue]]; // Update page thumb view
259 
260  NSInteger strideThumbs = (thumbs - 1); if (strideThumbs < 1) strideThumbs = 1;
261 
262  CGFloat stride = ((CGFloat)pages / (CGFloat)strideThumbs); // Page stride
263 
264  CGFloat heightDelta = (controlRect.size.height - THUMB_SMALL_HEIGHT);
265 
266  NSInteger thumbY = (heightDelta / 2.0f); NSInteger thumbX = 0; // Initial X, Y
267 
268  CGRect thumbRect = CGRectMake(thumbX, thumbY, THUMB_SMALL_WIDTH, THUMB_SMALL_HEIGHT);
269 
270  NSMutableDictionary *thumbsToHide = [miniThumbViews mutableCopy];
271 
272  for (NSInteger thumb = 0; thumb < thumbs; thumb++) // Iterate through needed thumbs
273  {
274  NSInteger page = ((stride * thumb) + 1); if (page > pages) page = pages; // Page
275 
276  NSNumber *key = [NSNumber numberWithInteger:page]; // Page number key for thumb view
277 
278  ReaderPagebarThumb *smallThumbView = [miniThumbViews objectForKey:key]; // Thumb view
279 
280  if (smallThumbView == nil) // We need to create a new small thumb view for the page number
281  {
282  CGSize size = CGSizeMake(THUMB_SMALL_WIDTH, THUMB_SMALL_HEIGHT); // Maximum thumb size
283 
284  NSURL *fileURL = document.fileURL; NSString *guid = document.guid; NSString *phrase = document.password;
285 
286  smallThumbView = [[ReaderPagebarThumb alloc] initWithFrame:thumbRect small:YES]; // Create a small thumb view
287 
288  ReaderThumbRequest *thumbRequest = [ReaderThumbRequest newForView:smallThumbView fileURL:fileURL password:phrase guid:guid page:page size:size];
289 
290  UIImage *image = [[ReaderThumbCache sharedInstance] thumbRequest:thumbRequest priority:NO]; // Request the thumb
291 
292  if ([image isKindOfClass:[UIImage class]]) [smallThumbView showImage:image]; // Use thumb image from cache
293 
294  [trackControl addSubview:smallThumbView]; [miniThumbViews setObject:smallThumbView forKey:key];
295  }
296  else // Resue existing small thumb view for the page number
297  {
298  smallThumbView.hidden = NO; [thumbsToHide removeObjectForKey:key];
299 
300  if (CGRectEqualToRect(smallThumbView.frame, thumbRect) == false)
301  {
302  smallThumbView.frame = thumbRect; // Update thumb frame
303  }
304  }
305 
306  thumbRect.origin.x += thumbWidth; // Next thumb X position
307  }
308 
309  // Hide unused thumbs[thumbsToHide enumerateKeysAndObjectsUsingBlock:
310  ^(id key, id object, BOOL *stop)
311  {
312  ReaderPagebarThumb *thumb = object; thumb.hidden = YES;
313  }
314  ];
315 }
