65 static inline CGFloat ZoomScaleThatFits(CGSize target, CGSize source)
66 {
67  CGFloat w_scale = (target.width / source.width);
68 
69  CGFloat h_scale = (target.height / source.height);
70 
71  return ((w_scale < h_scale) ? w_scale : h_scale);
72 }

76 - (void)updateMinimumMaximumZoom
77 {
78  CGRect targetRect = CGRectInset(self.bounds, CONTENT_INSET, CONTENT_INSET);
79 
80  CGFloat zoomScale = ZoomScaleThatFits(targetRect.size, theContentView.bounds.size);
81 
82  self.minimumZoomScale = zoomScale; // Set the minimum and maximum zoom scales
83 
84  self.maximumZoomScale = (zoomScale * ZOOM_LEVELS); // Max number of zoom levels
85 
86  zoomAmount = ((self.maximumZoomScale - self.minimumZoomScale) / ZOOM_LEVELS);
87 }

178 - (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
179 {
180  if (context == ReaderContentViewContext) // Our context
181  {
182  if ((object == self) && [keyPath isEqualToString:@"frame"])
183  {
184  CGFloat oldMinimumZoomScale = self.minimumZoomScale;
185 
186  [self updateMinimumMaximumZoom]; // Update zoom scale limits
187 
188  if (self.zoomScale == oldMinimumZoomScale) // Old minimum
189  {
190  self.zoomScale = self.minimumZoomScale;
191  }
192  else // Check against minimum zoom scale
193  {
194  if (self.zoomScale < self.minimumZoomScale)
195  {
196  self.zoomScale = self.minimumZoomScale;
197  }
198  else // Check against maximum zoom scale
199  {
200  if (self.zoomScale > self.maximumZoomScale)
201  {
202  self.zoomScale = self.maximumZoomScale;
203  }
204  }
205  }
206  }
207  }
