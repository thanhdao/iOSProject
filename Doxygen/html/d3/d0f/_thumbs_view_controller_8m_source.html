197 - (void)tappedInToolbar:(ThumbsMainToolbar *)toolbar showControl:(UISegmentedControl *)control
198 {
199  switch (control.selectedSegmentIndex)
200  {
201  case 0: // Show all page thumbs
202  {
203  showBookmarked = NO; // Show all thumbs
204 
205  markedOffset = [theThumbsView insetContentOffset];
206 
207  [theThumbsView reloadThumbsContentOffset:thumbsOffset];
208 
209  break; // We're done
210  }
211 
212  case 1: // Show bookmarked thumbs
213  {
214  showBookmarked = YES; // Only bookmarked
215 
216  thumbsOffset = [theThumbsView insetContentOffset];
217 
218  if (updateBookmarked == YES) // Update bookmarked list
219  {
220  [bookmarked removeAllObjects]; // Empty the list first
221 
222  // Enumerate[document.bookmarks enumerateIndexesUsingBlock:
223  ^(NSUInteger page, BOOL *stop)
224  {
225  [bookmarked addObject:[NSNumber numberWithInteger:page]];
226  }
227  ];
228 
229  markedOffset = CGPointZero; updateBookmarked = NO; // Reset
230  }
231 
232  [theThumbsView reloadThumbsContentOffset:markedOffset];
233 
234  break; // We're done
235  }
236  }
237 }

256 - (void)thumbsView:(ReaderThumbsView *)thumbsView updateThumbCell:(ThumbsPageThumb *)thumbCell forIndex:(NSInteger)index
257 {
258  CGSize size = [thumbCell maximumContentSize]; // Get the cell's maximum content size
259 
260  NSInteger page = (showBookmarked ? [[bookmarked objectAtIndex:index] integerValue] : (index + 1));
261 
262  [thumbCell showText:[NSString stringWithFormat:@"%d", page]]; // Page number place holder
263 
264  [thumbCell showBookmark:[document.bookmarks containsIndex:page]]; // Show bookmarked status
265 
266  NSURL *fileURL = document.fileURL; NSString *guid = document.guid; NSString *phrase = document.password; // Document info
267 
268  ReaderThumbRequest *thumbRequest = [ReaderThumbRequest newForView:thumbCell fileURL:fileURL password:phrase guid:guid page:page size:size];
269 
270  UIImage *image = [[ReaderThumbCache sharedInstance] thumbRequest:thumbRequest priority:YES]; // Request the thumbnail
271 
272  if ([image isKindOfClass:[UIImage class]]) [thumbCell showImage:image]; // Show image from cache
273 }
