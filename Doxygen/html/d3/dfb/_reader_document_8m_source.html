170 + (BOOL)isPDF:(NSString *)filePath
171 {
172  BOOL state = NO;
173 
174  if (filePath != nil) // Must have a file path
175  {
176  const char *path = [filePath fileSystemRepresentation];
177 
178  int fd = open(path, O_RDONLY); // Open the file
179 
180  if (fd > 0) // We have a valid file descriptor
181  {
182  const char sig[1024]; // File signature buffer
183 
184  ssize_t len = read(fd, (void *)&sig, sizeof(sig));
185 
186  state = (strnstr(sig, "%PDF", len) != NULL);
187 
188  close(fd); // Close the file
189  }
190  }
191 
192  return state;
193 }
