78 - (ReaderDocumentLink *)linkFromAnnotation:(CGPDFDictionaryRef)annotationDictionary
79 {
80  ReaderDocumentLink *documentLink = nil; // Document link object
81 
82  CGPDFArrayRef annotationRectArray = NULL; // Annotation co-ordinates array
83 
84  if (CGPDFDictionaryGetArray(annotationDictionary, "Rect", &annotationRectArray))
85  {
86  CGPDFReal ll_x = 0.0f; CGPDFReal ll_y = 0.0f; // PDFRect lower-left X and Y
87  CGPDFReal ur_x = 0.0f; CGPDFReal ur_y = 0.0f; // PDFRect upper-right X and Y
88 
89  CGPDFArrayGetNumber(annotationRectArray, 0, &ll_x); // Lower-left X co-ordinate
90  CGPDFArrayGetNumber(annotationRectArray, 1, &ll_y); // Lower-left Y co-ordinate
91 
92  CGPDFArrayGetNumber(annotationRectArray, 2, &ur_x); // Upper-right X co-ordinate
93  CGPDFArrayGetNumber(annotationRectArray, 3, &ur_y); // Upper-right Y co-ordinate
94 
95  if (ll_x > ur_x) { CGPDFReal t = ll_x; ll_x = ur_x; ur_x = t; } // Normalize Xs
96  if (ll_y > ur_y) { CGPDFReal t = ll_y; ll_y = ur_y; ur_y = t; } // Normalize Ys
97 
98  ll_x -= _pageOffsetX; ll_y -= _pageOffsetY; // Offset lower-left co-ordinate
99  ur_x -= _pageOffsetX; ur_y -= _pageOffsetY; // Offset upper-right co-ordinate
100 
101  switch (_pageAngle) // Page rotation angle (in degrees)
102  {
103  case 90: // 90 degree page rotation
104  {
105  CGPDFReal swap;
106  swap = ll_y; ll_y = ll_x; ll_x = swap;
107  swap = ur_y; ur_y = ur_x; ur_x = swap;
108  break;
109  }
110 
111  case 270: // 270 degree page rotation
112  {
113  CGPDFReal swap;
114  swap = ll_y; ll_y = ll_x; ll_x = swap;
115  swap = ur_y; ur_y = ur_x; ur_x = swap;
116  ll_x = ((0.0f - ll_x) + _pageWidth);
117  ur_x = ((0.0f - ur_x) + _pageWidth);
118  break;
119  }
120 
121  case 0: // 0 degree page rotation
122  {
123  ll_y = ((0.0f - ll_y) + _pageHeight);
124  ur_y = ((0.0f - ur_y) + _pageHeight);
125  break;
126  }
127  }
128 
129  NSInteger vr_x = ll_x; NSInteger vr_w = (ur_x - ll_x); // Integer X and width
130  NSInteger vr_y = ll_y; NSInteger vr_h = (ur_y - ll_y); // Integer Y and height
131 
132  CGRect viewRect = CGRectMake(vr_x, vr_y, vr_w, vr_h); // View CGRect from PDFRect
133 
134  documentLink = [ReaderDocumentLink newWithRect:viewRect dictionary:annotationDictionary];
135  }
136 
137  return documentLink;
138 }

176 - (CGPDFArrayRef)destinationWithName:(const char *)destinationName inDestsTree:(CGPDFDictionaryRef)node
177 {
178  CGPDFArrayRef destinationArray = NULL;
179 
180  CGPDFArrayRef limitsArray = NULL; // Limits array
181 
182  if (CGPDFDictionaryGetArray(node, "Limits", &limitsArray) == true)
183  {
184  CGPDFStringRef lowerLimit = NULL; CGPDFStringRef upperLimit = NULL;
185 
186  if (CGPDFArrayGetString(limitsArray, 0, &lowerLimit) == true) // Lower limit
187  {
188  if (CGPDFArrayGetString(limitsArray, 1, &upperLimit) == true) // Upper limit
189  {
190  const char *ll = (const char *)CGPDFStringGetBytePtr(lowerLimit); // Lower string
191  const char *ul = (const char *)CGPDFStringGetBytePtr(upperLimit); // Upper string
192 
193  if ((strcmp(destinationName, ll) < 0) || (strcmp(destinationName, ul) > 0))
194  {
195  return NULL; // Destination name is outside this node's limits
196  }
197  }
198  }
199  }
200 
201  CGPDFArrayRef namesArray = NULL; // Names array
202 
203  if (CGPDFDictionaryGetArray(node, "Names", &namesArray) == true)
204  {
205  NSInteger namesCount = CGPDFArrayGetCount(namesArray);
206 
207  for (NSInteger index = 0; index < namesCount; index += 2)
208  {
209  CGPDFStringRef destName; // Destination name string
210 
211  if (CGPDFArrayGetString(namesArray, index, &destName) == true)
212  {
213  const char *dn = (const char *)CGPDFStringGetBytePtr(destName);
214 
215  if (strcmp(dn, destinationName) == 0) // Found the destination name
216  {
217  if (CGPDFArrayGetArray(namesArray, (index + 1), &destinationArray) == false)
218  {
219  CGPDFDictionaryRef destinationDictionary = NULL; // Destination dictionary
220 
221  if (CGPDFArrayGetDictionary(namesArray, (index + 1), &destinationDictionary) == true)
222  {
223  CGPDFDictionaryGetArray(destinationDictionary, "D", &destinationArray);
224  }
225  }
226 
227  return destinationArray; // Return the destination array
228  }
229  }
230  }
231  }
232 
233  CGPDFArrayRef kidsArray = NULL; // Kids array
234 
235  if (CGPDFDictionaryGetArray(node, "Kids", &kidsArray) == true)
236  {
237  NSInteger kidsCount = CGPDFArrayGetCount(kidsArray);
238 
239  for (NSInteger index = 0; index < kidsCount; index++)
240  {
241  CGPDFDictionaryRef kidNode = NULL; // Kid node dictionary
242 
243  if (CGPDFArrayGetDictionary(kidsArray, index, &kidNode) == true) // Recurse into node
244  {
245  destinationArray = [self destinationWithName:destinationName inDestsTree:kidNode];
246 
247  if (destinationArray != NULL) return destinationArray; // Return destination array
248  }
249  }
250  }
251 
252  return NULL;
253 }

255 - (id)annotationLinkTarget:(CGPDFDictionaryRef)annotationDictionary
256 {
257  id linkTarget = nil; // Link target object
258 
259  CGPDFStringRef destName = NULL; const char *destString = NULL;
260 
261  CGPDFDictionaryRef actionDictionary = NULL; CGPDFArrayRef destArray = NULL;
262 
263  if (CGPDFDictionaryGetDictionary(annotationDictionary, "A", &actionDictionary) == true)
264  {
265  const char *actionType = NULL; // Annotation action type string
266 
267  if (CGPDFDictionaryGetName(actionDictionary, "S", &actionType) == true)
268  {
269  if (strcmp(actionType, "GoTo") == 0) // GoTo action type
270  {
271  if (CGPDFDictionaryGetArray(actionDictionary, "D", &destArray) == false)
272  {
273  CGPDFDictionaryGetString(actionDictionary, "D", &destName);
274  }
275  }
276  else // Handle other link action type possibility
277  {
278  if (strcmp(actionType, "URI") == 0) // URI action type
279  {
280  CGPDFStringRef uriString = NULL; // Action's URI string
281 
282  if (CGPDFDictionaryGetString(actionDictionary, "URI", &uriString) == true)
283  {
284  const char *uri = (const char *)CGPDFStringGetBytePtr(uriString); // Destination URI string
285 
286  linkTarget = [NSURL URLWithString:[NSString stringWithCString:uri encoding:NSASCIIStringEncoding]];
287  }
288  }
289  }
290  }
291  }
292  else // Handle other link target possibilities
293  {
294  if (CGPDFDictionaryGetArray(annotationDictionary, "Dest", &destArray) == false)
295  {
296  if (CGPDFDictionaryGetString(annotationDictionary, "Dest", &destName) == false)
297  {
298  CGPDFDictionaryGetName(annotationDictionary, "Dest", &destString);
299  }
300  }
301  }
302 
303  if (destName != NULL) // Handle a destination name
304  {
305  CGPDFDictionaryRef catalogDictionary = CGPDFDocumentGetCatalog(_PDFDocRef);
306 
307  CGPDFDictionaryRef namesDictionary = NULL; // Destination names in the document
308 
309  if (CGPDFDictionaryGetDictionary(catalogDictionary, "Names", &namesDictionary) == true)
310  {
311  CGPDFDictionaryRef destsDictionary = NULL; // Document destinations dictionary
312 
313  if (CGPDFDictionaryGetDictionary(namesDictionary, "Dests", &destsDictionary) == true)
314  {
315  const char *destinationName = (const char *)CGPDFStringGetBytePtr(destName); // Name
316 
317  destArray = [self destinationWithName:destinationName inDestsTree:destsDictionary];
318  }
319  }
320  }
321 
322  if (destString != NULL) // Handle a destination string
323  {
324  CGPDFDictionaryRef catalogDictionary = CGPDFDocumentGetCatalog(_PDFDocRef);
325 
326  CGPDFDictionaryRef destsDictionary = NULL; // Document destinations dictionary
327 
328  if (CGPDFDictionaryGetDictionary(catalogDictionary, "Dests", &destsDictionary) == true)
329  {
330  CGPDFDictionaryRef targetDictionary = NULL; // Destination target dictionary
331 
332  if (CGPDFDictionaryGetDictionary(destsDictionary, destString, &targetDictionary) == true)
333  {
334  CGPDFDictionaryGetArray(targetDictionary, "D", &destArray);
335  }
336  }
337  }
338 
339  if (destArray != NULL) // Handle a destination array
340  {
341  NSInteger targetPageNumber = 0; // The target page number
342 
343  CGPDFDictionaryRef pageDictionaryFromDestArray = NULL; // Target reference
344 
345  if (CGPDFArrayGetDictionary(destArray, 0, &pageDictionaryFromDestArray) == true)
346  {
347  NSInteger pageCount = CGPDFDocumentGetNumberOfPages(_PDFDocRef); // Pages
348 
349  for (NSInteger pageNumber = 1; pageNumber <= pageCount; pageNumber++)
350  {
351  CGPDFPageRef pageRef = CGPDFDocumentGetPage(_PDFDocRef, pageNumber);
352 
353  CGPDFDictionaryRef pageDictionaryFromPage = CGPDFPageGetDictionary(pageRef);
354 
355  if (pageDictionaryFromPage == pageDictionaryFromDestArray) // Found it
356  {
357  targetPageNumber = pageNumber; break;
358  }
359  }
360  }
361  else // Try page number from array possibility
362  {
363  CGPDFInteger pageNumber = 0; // Page number in array
364 
365  if (CGPDFArrayGetInteger(destArray, 0, &pageNumber) == true)
366  {
367  targetPageNumber = (pageNumber + 1); // 1-based
368  }
369  }
370 
371  if (targetPageNumber > 0) // We have a target page number
372  {
373  linkTarget = [NSNumber numberWithInteger:targetPageNumber];
374  }
375  }
376 
377  return linkTarget;
378 }

430 - (id)initWithURL:(NSURL *)fileURL page:(NSInteger)page password:(NSString *)phrase
431 {
432  CGRect viewRect = CGRectZero; // View rect
433 
434  if (fileURL != nil) // Check for non-nil file URL
435  {
436  _PDFDocRef = CGPDFDocumentCreateX((__bridge CFURLRef)fileURL, phrase);
437 
438  if (_PDFDocRef != NULL) // Check for non-NULL CGPDFDocumentRef
439  {
440  if (page < 1) page = 1; // Check the lower page bounds
441 
442  NSInteger pages = CGPDFDocumentGetNumberOfPages(_PDFDocRef);
443 
444  if (page > pages) page = pages; // Check the upper page bounds
445 
446  _PDFPageRef = CGPDFDocumentGetPage(_PDFDocRef, page); // Get page
447 
448  if (_PDFPageRef != NULL) // Check for non-NULL CGPDFPageRef
449  {
450  CGPDFPageRetain(_PDFPageRef); // Retain the PDF page
451 
452  CGRect cropBoxRect = CGPDFPageGetBoxRect(_PDFPageRef, kCGPDFCropBox);
453  CGRect mediaBoxRect = CGPDFPageGetBoxRect(_PDFPageRef, kCGPDFMediaBox);
454  CGRect effectiveRect = CGRectIntersection(cropBoxRect, mediaBoxRect);
455 
456  _pageAngle = CGPDFPageGetRotationAngle(_PDFPageRef); // Angle
457 
458  switch (_pageAngle) // Page rotation angle (in degrees)
459  {
460  default: // Default case
461  case 0: case 180: // 0 and 180 degrees
462  {
463  _pageWidth = effectiveRect.size.width;
464  _pageHeight = effectiveRect.size.height;
465  _pageOffsetX = effectiveRect.origin.x;
466  _pageOffsetY = effectiveRect.origin.y;
467  break;
468  }
469 
470  case 90: case 270: // 90 and 270 degrees
471  {
472  _pageWidth = effectiveRect.size.height;
473  _pageHeight = effectiveRect.size.width;
474  _pageOffsetX = effectiveRect.origin.y;
475  _pageOffsetY = effectiveRect.origin.x;
476  break;
477  }
478  }
479 
480  NSInteger page_w = _pageWidth; // Integer width
481  NSInteger page_h = _pageHeight; // Integer height
482 
483  if (page_w % 2) page_w--; if (page_h % 2) page_h--; // Even
484 
485  viewRect.size = CGSizeMake(page_w, page_h); // View size
486  }
487  else // Error out with a diagnostic
488  {
489  CGPDFDocumentRelease(_PDFDocRef), _PDFDocRef = NULL;
490 
491  NSAssert(NO, @"CGPDFPageRef == NULL");
492  }
493  }
494  else // Error out with a diagnostic
495  {
496  NSAssert(NO, @"CGPDFDocumentRef == NULL");
497  }
498  }
499  else // Error out with a diagnostic
500  {
501  NSAssert(NO, @"fileURL == nil");
502  }
503 
504  id view = [self initWithFrame:viewRect]; // UIView setup
505 
506  if (view != nil) [self buildAnnotationLinksList]; // Links
507 
508  return view;
509 }
