37 void logDictionaryEntry(const char *key, CGPDFObjectRef object, void *info)

122 + (CGPDFArrayRef)destinationWithName:(const char *)destinationName inDestsTree:(CGPDFDictionaryRef)node
123 {
124  CGPDFArrayRef destinationArray = NULL;
125 
126  CGPDFArrayRef limitsArray = NULL; // Limits array
127 
128  if (CGPDFDictionaryGetArray(node, "Limits", &limitsArray) == true)
129  {
130  CGPDFStringRef lowerLimit = NULL; CGPDFStringRef upperLimit = NULL;
131 
132  if (CGPDFArrayGetString(limitsArray, 0, &lowerLimit) == true) // Lower limit
133  {
134  if (CGPDFArrayGetString(limitsArray, 1, &upperLimit) == true) // Upper limit
135  {
136  const char *ll = (const char *)CGPDFStringGetBytePtr(lowerLimit); // Lower string
137  const char *ul = (const char *)CGPDFStringGetBytePtr(upperLimit); // Upper string
138 
139  if ((strcmp(destinationName, ll) < 0) || (strcmp(destinationName, ul) > 0))
140  {
141  return NULL; // Destination name is outside this node's limits
142  }
143  }
144  }
145  }
146 
147  CGPDFArrayRef namesArray = NULL; // Names array
148 
149  if (CGPDFDictionaryGetArray(node, "Names", &namesArray) == true)
150  {
151  NSInteger namesCount = CGPDFArrayGetCount(namesArray);
152 
153  for (NSInteger index = 0; index < namesCount; index += 2)
154  {
155  CGPDFStringRef destName; // Destination name string
156 
157  if (CGPDFArrayGetString(namesArray, index, &destName) == true)
158  {
159  const char *dn = (const char *)CGPDFStringGetBytePtr(destName);
160 
161  if (strcmp(dn, destinationName) == 0) // Found the destination name
162  {
163  if (CGPDFArrayGetArray(namesArray, (index + 1), &destinationArray) == false)
164  {
165  CGPDFDictionaryRef destinationDictionary = NULL; // Destination dictionary
166 
167  if (CGPDFArrayGetDictionary(namesArray, (index + 1), &destinationDictionary) == true)
168  {
169  CGPDFDictionaryGetArray(destinationDictionary, "D", &destinationArray);
170  }
171  }
172 
173  return destinationArray; // Return the destination array
174  }
175  }
176  }
177  }
178 
179  CGPDFArrayRef kidsArray = NULL; // Kids array
180 
181  if (CGPDFDictionaryGetArray(node, "Kids", &kidsArray) == true)
182  {
183  NSInteger kidsCount = CGPDFArrayGetCount(kidsArray);
184 
185  for (NSInteger index = 0; index < kidsCount; index++)
186  {
187  CGPDFDictionaryRef kidNode = NULL; // Kid node dictionary
188 
189  if (CGPDFArrayGetDictionary(kidsArray, index, &kidNode) == true) // Recurse into node
190  {
191  destinationArray = [self destinationWithName:destinationName inDestsTree:kidNode];
192 
193  if (destinationArray != NULL) return destinationArray; // Return destination array
194  }
195  }
196  }
197 
198  return NULL;
199 }

201 + (id)outlineEntryTarget:(CGPDFDictionaryRef)outlineDictionary document:(CGPDFDocumentRef)document
202 {
203  id entryTarget = nil; // Entry target object
204 
205  CGPDFStringRef destName = NULL; const char *destString = NULL;
206 
207  CGPDFDictionaryRef actionDictionary = NULL; CGPDFArrayRef destArray = NULL;
208 
209  if (CGPDFDictionaryGetDictionary(outlineDictionary, "A", &actionDictionary) == true)
210  {
211  const char *actionType = NULL; // Outline entry action type string
212 
213  if (CGPDFDictionaryGetName(actionDictionary, "S", &actionType) == true)
214  {
215  if (strcmp(actionType, "GoTo") == 0) // GoTo action type
216  {
217  if (CGPDFDictionaryGetArray(actionDictionary, "D", &destArray) == false)
218  {
219  CGPDFDictionaryGetString(actionDictionary, "D", &destName);
220  }
221  }
222  else // Handle other entry action type possibility
223  {
224  if (strcmp(actionType, "URI") == 0) // URI action type
225  {
226  CGPDFStringRef uriString = NULL; // Action's URI string
227 
228  if (CGPDFDictionaryGetString(actionDictionary, "URI", &uriString) == true)
229  {
230  const char *uri = (const char *)CGPDFStringGetBytePtr(uriString); // Destination URI string
231 
232  entryTarget = [NSURL URLWithString:[NSString stringWithCString:uri encoding:NSASCIIStringEncoding]];
233  }
234  }
235  }
236  }
237  }
238  else // Handle other entry target possibilities
239  {
240  if (CGPDFDictionaryGetArray(outlineDictionary, "Dest", &destArray) == false)
241  {
242  if (CGPDFDictionaryGetString(outlineDictionary, "Dest", &destName) == false)
243  {
244  CGPDFDictionaryGetName(outlineDictionary, "Dest", &destString);
245  }
246  }
247  }
248 
249  if (destName != NULL) // Handle a destination name
250  {
251  CGPDFDictionaryRef catalogDictionary = CGPDFDocumentGetCatalog(document);
252 
253  CGPDFDictionaryRef namesDictionary = NULL; // Destination names in the document
254 
255  if (CGPDFDictionaryGetDictionary(catalogDictionary, "Names", &namesDictionary) == true)
256  {
257  CGPDFDictionaryRef destsDictionary = NULL; // Document destinations dictionary
258 
259  if (CGPDFDictionaryGetDictionary(namesDictionary, "Dests", &destsDictionary) == true)
260  {
261  const char *destinationName = (const char *)CGPDFStringGetBytePtr(destName); // Name
262 
263  destArray = [self destinationWithName:destinationName inDestsTree:destsDictionary];
264  }
265  }
266  }
267 
268  if (destString != NULL) // Handle a destination string
269  {
270  CGPDFDictionaryRef catalogDictionary = CGPDFDocumentGetCatalog(document);
271 
272  CGPDFDictionaryRef destsDictionary = NULL; // Document destinations dictionary
273 
274  if (CGPDFDictionaryGetDictionary(catalogDictionary, "Dests", &destsDictionary) == true)
275  {
276  CGPDFDictionaryRef targetDictionary = NULL; // Destination target dictionary
277 
278  if (CGPDFDictionaryGetDictionary(destsDictionary, destString, &targetDictionary) == true)
279  {
280  CGPDFDictionaryGetArray(targetDictionary, "D", &destArray);
281  }
282  }
283  }
284 
285  if (destArray != NULL) // Handle a destination array
286  {
287  NSInteger targetPageNumber = 0; // The target page number
288 
289  CGPDFDictionaryRef pageDictionaryFromDestArray = NULL; // Target reference
290 
291  if (CGPDFArrayGetDictionary(destArray, 0, &pageDictionaryFromDestArray) == true)
292  {
293  NSInteger pageCount = CGPDFDocumentGetNumberOfPages(document); // Pages
294 
295  for (NSInteger pageNumber = 1; pageNumber <= pageCount; pageNumber++)
296  {
297  CGPDFPageRef pageRef = CGPDFDocumentGetPage(document, pageNumber);
298 
299  CGPDFDictionaryRef pageDictionaryFromPage = CGPDFPageGetDictionary(pageRef);
300 
301  if (pageDictionaryFromPage == pageDictionaryFromDestArray) // Found it
302  {
303  targetPageNumber = pageNumber; break;
304  }
305  }
306  }
307  else // Try page number from array possibility
308  {
309  CGPDFInteger pageNumber = 0; // Page number in array
310 
311  if (CGPDFArrayGetInteger(destArray, 0, &pageNumber) == true)
312  {
313  targetPageNumber = (pageNumber + 1); // 1-based
314  }
315  }
316 
317  if (targetPageNumber > 0) // We have a target page number
318  {
319  entryTarget = [NSNumber numberWithInteger:targetPageNumber];
320  }
321  }
322 
323  return entryTarget;
324 }

326 + (void)outlineItems:(CGPDFDictionaryRef)outlineDictionary document:(CGPDFDocumentRef)document array:(NSMutableArray *)array level:(NSInteger)level
327 {
328  do // Loop through current level outline entries
329  {
330  DocumentOutlineEntry *outlineEntry = nil; // An entry
331 
332  CGPDFStringRef string = NULL; // Outline entry title string
333 
334  if (CGPDFDictionaryGetString(outlineDictionary, "Title", &string) == true)
335  {
336  CFStringRef title = NULL; // Actual outline title (CFObject) string
337 
338  if ((title = CGPDFStringCopyTextString(string)) != NULL) // Copy of CFObject string
339  {
340  NSString *titleString = (__bridge NSString *)title; // CFString to NSString toll-free bridge cast
341 
342  id entryTarget = [self outlineEntryTarget:outlineDictionary document:document]; // Get target object
343 
344  NSString *trimmed = [titleString stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
345 
346  outlineEntry = [DocumentOutlineEntry newWithTitle:trimmed target:entryTarget level:level]; // New entry
347 
348  [array addObject:outlineEntry]; CFRelease(title); // Add new entry and cleanup
349  }
350  }
351 
352  if (outlineEntry != nil) // Must have a current outline entry
353  {
354  CGPDFDictionaryRef childItem = NULL; // First child outline item
355 
356  if (CGPDFDictionaryGetDictionary(outlineDictionary, "First", &childItem) == true)
357  {
358 #if (HIERARCHICAL_OUTLINE == TRUE) // Option
359  NSMutableArray *childArray = [NSMutableArray array]; outlineEntry.children = childArray;
360 
361  [self outlineItems:childItem document:document array:childArray level:(level + 1)];
362 #else // Flat
363  [self outlineItems:childItem document:document array:array level:(level + 1)];
364 
365 #endif // end of HIERARCHICAL_OUTLINE Option
366  }
367  }
368 
369  } while (CGPDFDictionaryGetDictionary(outlineDictionary, "Next", &outlineDictionary) == true);
370 }
